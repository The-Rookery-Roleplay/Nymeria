	unique = yes
	ai_score = {
		value = 100

					any_artifact = { has_variable = oathkeeper_artifact }
				}
			}
			add = -100
		}
	}
}

asoiaf_aerys_II_crown_template = {
	can_equip = {
		agot_can_wear_crown_trigger = yes
	}

	# can this character benefit from the full modifiers of the artifact?
	can_benefit = {
		always = yes
	}

	# if a given character does not pass the "can_benefit" trigger then this modifier will be applied instead.
	fallback = {
	}

	unique = yes
	ai_score = {
		value = 101 #1 more than the vanilla AGOT Stannis crown
		if = { #if you're not Valyrian or Rhllorist, then don't wear it
			limit = {
				NOR = {
					culture = { has_cultural_pillar = heritage_valyrian }
					religion = religion:the_rhllor_religion
				}
			}
			add = -10000
		}
		if = {
			limit = {
				any_equipped_character_artifact = {
					has_variable = stannis_crown_artifact
					NOT = { has_variable = asoiaf_aerys_II_crown_artefact }
				}
			}
			add = 100000 #In AGOT+'s modcanon, Stannis wears Aerys II's crown, not vanilla AGOT's crown
		}
	}
}
