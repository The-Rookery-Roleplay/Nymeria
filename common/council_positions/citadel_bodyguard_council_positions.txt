### Citadel Bodyguard orders (council positions) - duplicated from AGOT kingsguard for Vale (Winged Knights) + Reach (Green Knights). - Kizan

##### WINGED KNIGHTS (Vale) #####
winged_knight_lord_commander = { # Special
	skill = prowess

	name = {
		first_valid = {
			triggered_desc = {
				trigger = {
					liege ?= {
						ruler_primary_title_has_winged_knight_trigger = yes
						is_male = yes
					}
				}
				desc = winged_knight_lord_commander_name
			}
			triggered_desc = {
				trigger = {
					liege ?= {
						ruler_primary_title_has_winged_knight_trigger = yes
						is_female = yes
					}
				}
				desc = winged_knight_lord_commander_name
			}
		}
	}

	# Give public opinion and control tasks

	tooltip = game_concept_lord_commander_desc

	modifier = {
		name = winged_knight_lord_commander_modifier
		monthly_prestige = 1
		monthly_dynasty_prestige = 1
	}

	can_fire = { # A winged_knight serves for life; Joffrey interrupted that tradition;
		custom_description = {
			text = "agot_kingsguard_can_fire"
			OR = {
				has_global_variable = agot_startup_fire_councillors
				cp:winged_knight_lord_commander = { has_character_flag = can_fire_winged_knight }
				# has_realm_law = winged_knight_law_2
			}
		}
	}

	can_reassign = {
		custom_description = {
			text = "agot_kingsguard_can_fire"
			OR = {
				has_global_variable = agot_startup_fire_councillors
				cp:winged_knight_lord_commander ?= { has_character_flag = can_fire_winged_knight }
				# has_realm_law = winged_knight_law_2
			}
		}
	}

	valid_position = {
		can_have_winged_knight = yes
		#has_character_flag = assigning_winged_knight This will be useful soon
	}

	valid_character = {
		exists = root.liege_or_court_owner
		can_be_winged_knight_trigger = { COURT_OWNER = root.liege_or_court_owner }
		has_trait = winged_knight
	}

	on_get_position = {
		agot_on_get_winged_knight_lord_commander = yes
	}

	on_lose_position = {
		agot_on_fired_from_winged_knight = yes
	}

	on_fired_from_position = {
		agot_on_fired_from_winged_knight = yes
		#remove_character_flag = lord_commander Need to keep this for weirdness
	}
	portrait_animation = kingsguard
}
