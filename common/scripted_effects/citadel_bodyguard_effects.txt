##### WINGED KNIGHTS (Vale) #####
agot_join_winged_knight_effect = {
	$KINGSGUARD$ = { save_scope_as = winged_knight }
	$KING$ = { save_scope_as = king }
	scope:winged_knight = {
		add_trait = winged_knight
		add_character_flag = { flag = blocked_from_leaving }
		if = {
			limit = {
				has_trait = refusing_marriage
			}
			remove_trait = refusing_marriage
		}
		if = {
			limit = {
				has_character_modifier = training_for_winged_knight
			}
			remove_character_modifier = training_for_winged_knight
		}
	}
	scope:king = {
		if = {
			limit = {
				NAND = {
					exists = scope:winged_knight.liege_or_court_owner
					scope:winged_knight.liege_or_court_owner = this
				}
			}
			add_courtier ?= scope:winged_knight
		}
		if = {
			limit = {
				scope:winged_knight = {
					has_character_flag = lord_commander
				}
			}
			if = {
				limit = {
					NOT = {
						any_councillor = {
							has_council_position = winged_knight_lord_commander
						}
					}
				}
				assign_councillor_type = {
					type = winged_knight_lord_commander
					target = scope:winged_knight
				}
			}
			primary_title = {
				set_variable = {
					name = winged_knight_lord_commander
					value = scope:winged_knight
				}
			}
		}
		else_if = {
			limit = {
				NOT = { exists = cp:winged_knight_1 }
			}
			assign_councillor_type = {
				type = winged_knight_1
				target = scope:winged_knight
			}
			primary_title = {
				set_variable = {
					name = winged_knight_1
					value = scope:winged_knight
				}
			}
		}
		else_if = {
			limit = {
				NOT = { exists = cp:winged_knight_2 }
			}
			assign_councillor_type = {
				type = winged_knight_2
				target = scope:winged_knight
			}
			primary_title = {
				set_variable = {
					name = winged_knight_2
					value = scope:winged_knight
				}
			}
		}
		else_if = {
			limit = {
				NOT = { exists = cp:winged_knight_3 }
			}
			assign_councillor_type = {
				type = winged_knight_3
				target = scope:winged_knight
			}
			primary_title = {
				set_variable = {
					name = winged_knight_3
					value = scope:winged_knight
				}
			}
		}
		else_if = {
			limit = {
				NOT = { exists = cp:winged_knight_4 }
			}
			assign_councillor_type = {
				type = winged_knight_4
				target = scope:winged_knight
			}
			primary_title = {
				set_variable = {
					name = winged_knight_4
					value = scope:winged_knight
				}
			}
		}
		else_if = {
			limit = {
				NOT = { exists = cp:winged_knight_5 }
			}
			assign_councillor_type = {
				type = winged_knight_5
				target = scope:winged_knight
			}
			primary_title = {
				set_variable = {
					name = winged_knight_5
					value = scope:winged_knight
				}
			}
		}
		else_if = {
			limit = {
				NOT = { exists = cp:winged_knight_6 }
			}
			assign_councillor_type = {
				type = winged_knight_6
				target = scope:winged_knight
			}
			primary_title = {
				set_variable = {
					name = winged_knight_6
					value = scope:winged_knight
				}
			}
		}
	}
}

agot_assign_winged_knight_lord_commander_effect = {
	$COMMANDER$ = {
		create_character_memory = {
			type = agot_winged_knight_lord_commander_memory
			participants = {
				king = $KING$
			}
		}
		add_character_flag = lord_commander
	}

	$KING$ = {
		primary_title = {
			if = {
				limit = {
					exists = var:winged_knight_1
					var:winged_knight_1 = $COMMANDER$
				}
				remove_variable = winged_knight_1
			}
			else_if = {
				limit = {
					exists = var:winged_knight_2
					var:winged_knight_2 = $COMMANDER$
				}
				remove_variable = winged_knight_2
			}
			else_if = {
				limit = {
					exists = var:winged_knight_3
					var:winged_knight_3 = $COMMANDER$
				}
				remove_variable = winged_knight_3
			}
			else_if = {
				limit = {
					exists = var:winged_knight_4
					var:winged_knight_4 = $COMMANDER$
				}
				remove_variable = winged_knight_4
			}
			else_if = {
				limit = {
					exists = var:winged_knight_5
					var:winged_knight_5 = $COMMANDER$
				}
				remove_variable = winged_knight_5
			}
			else_if = {
				limit = {
					exists = var:winged_knight_6
					var:winged_knight_6 = $COMMANDER$
				}
				remove_variable = winged_knight_6
			}

			set_variable = {
				name = winged_knight_lord_commander
				value = $COMMANDER$
			}
		}


		assign_councillor_type = {
			type = winged_knight_lord_commander
			target = $COMMANDER$
		}
	}
	if = {
		limit = {
			$COMMANDER$ = { is_lowborn = yes }
		}
		save_temporary_scope_as = lc_is_lowborn
	}

	hidden_effect = {
		title:d_winged_knight = {
			copy_title_history = title:d_dummy_winged_knight
		}
	}

	create_title_and_vassal_change = {
		type = granted
		save_scope_as = change
	}
	title:d_winged_knight = {
		change_title_holder = {
			holder = $COMMANDER$
			change = scope:change
		}
	}

	resolve_title_and_vassal_change = scope:change

	if = {
		limit = {
			NOT = {
				$COMMANDER$.liege = $KING$
			}
		}
		create_title_and_vassal_change = {
			type = swear_fealty
			save_scope_as = fealty_change
			add_claim_on_loss = no
		}
		title:d_winged_knight.holder = {
			change_liege = {
				liege = $KING$
				change = scope:fealty_change
			}
		}
		resolve_title_and_vassal_change = scope:fealty_change
	}

	if = {
		limit = { exists = scope:lc_is_lowborn }
		$COMMANDER$ = { set_to_lowborn = yes }
	}

	title:d_winged_knight = {
		set_coa = d_winged_knight
	}

	hidden_effect = {
		title:d_dummy_winged_knight = {
			copy_title_history = title:d_winged_knight
		}
	}

	$KING$ = { trigger_event = agot_winged_knight.9007 } #grab back the lord commander you yeeted
}

agot_remove_winged_knight_effect = {
	$KINGSGUARD$ = { save_scope_as = removed_kg }

	scope:removed_kg = {
		if = {
			limit = {
				OR = {
					has_council_position = winged_knight_lord_commander
					has_character_flag = lord_commander
				}
			}
			save_scope_as = kg_is_lord_commander
		}
	}

	scope:removed_kg = {
		if = {
			limit = {
				NOT = { exists = scope:dead_winged_knight }
			}

			remove_trait = winged_knight
			add_character_modifier = agot_former_winged_knight
			agot_winged_knight_destroy_artifacts = yes
			add_character_flag = can_fire_winged_knight
			title:e_the_vale.holder ?= {
				if = {
					limit = {
						prev = { is_councillor_of = title:e_the_vale.holder }
					}
					fire_councillor = prev
				}
			}
		}
	}


	title:e_the_vale = {
		if = {
			limit = {
				exists = var:winged_knight_lord_commander
				var:winged_knight_lord_commander = scope:removed_kg
			}
			remove_variable = winged_knight_lord_commander
		}
		else_if = {
			limit = {
				exists = var:winged_knight_1
				var:winged_knight_1 = scope:removed_kg
			}
			remove_variable = winged_knight_1
		}
		else_if = {
			limit = {
				exists = var:winged_knight_2
				var:winged_knight_2 = scope:removed_kg
			}
			remove_variable = winged_knight_2
		}
		else_if = {
			limit = {
				exists = var:winged_knight_3
				var:winged_knight_3 = scope:removed_kg
			}
			remove_variable = winged_knight_3
		}
		else_if = {
			limit = {
				exists = var:winged_knight_4
				var:winged_knight_4 = scope:removed_kg
			}
			remove_variable = winged_knight_4
		}
		else_if = {
			limit = {
				exists = var:winged_knight_5
				var:winged_knight_5 = scope:removed_kg
			}
			remove_variable = winged_knight_5
		}
		else_if = {
			limit = {
				exists = var:winged_knight_6
				var:winged_knight_6 = scope:removed_kg
			}
			remove_variable = winged_knight_6
		}
	}

	title:e_the_vale.holder ?= {
		if = {
			limit = {
				exists = scope:kg_is_lord_commander
			}
			if = {
				limit = {
					NOT = { exists = scope:dead_winged_knight }
				}
				destroy_title = title:d_winged_knight
			}
			if = {
				limit = {
					any_councillor = {
						is_winged_knight_trigger = yes
						has_trait = winged_knight
						NOT = { this = scope:kg_is_lord_commander }
					}
				}
				if = {
					limit = {
						is_ai = yes
					}
					trigger_event = {
						id = agot_winged_knight.9001
						days = 3
					}
				}
				else = {
					trigger_event = {
						id = agot_winged_knight.1010
						delayed = yes
					}
				}
			}
		}

		if = {
			limit = {
				is_ai = yes
			}
			trigger_event = {
				id = agot_winged_knight.9002
				days = 4
			}
		}
		else_if = {
			limit = {
				is_ai = no
				NOT = {
					has_character_flag = skip_kg_replacement_event_flag
				}
			}
			trigger_event = agot_winged_knight.1001
		}
	}
}

agot_on_get_winged_knight = {
	got_council_position_effect = yes
	add_character_flag = blocked_from_leaving
	if = {
		limit = {
			exists = betrothed
		}
		if = {
			limit = {
				has_been_promised_grand_wedding = yes
			}
			break_grand_wedding_betrothal_effect = yes
		}
		else = {
			break_betrothal = betrothed
		}
	}
}

agot_winged_knight_fled_effect = {
	agot_remove_winged_knight_effect = { KINGSGUARD = scope:recipient }
	add_character_modifier = winged_knight_oathbreaker

	scope:recipient = {
		save_scope_as = fleeing_winged_knight
	}

	scope:actor = {
		save_scope_as = crown
	}

	scope:crown = {
		add_opinion = {
			target = scope:fleeing_winged_knight
			modifier = broke_vows_opinion
		}
	}

	if = {
		limit = {
			any_parent = {
				is_landed = yes
				NOT = {
					this = scope:crown
				}
			}
		}
		random_parent = {
			limit = {
				is_landed = yes
				NOT = {
					this = scope:crown
				}
			}
			save_scope_as = ruling_relative
		}
	}
	else_if = {
		limit = {
			any_close_family_member = {
				is_landed = yes
				NOT = {
					this = scope:crown
				}
			}
		}
		random_close_family_member = {
			limit = {
				is_landed = yes
				NOT = {
					this = scope:crown
				}
			}
			save_scope_as = ruling_relative
		}
	}
	else_if = {
		limit = {
			any_close_or_extended_family_member = {
				is_landed = yes
				NOT = {
					this = scope:crown
				}
			}
		}
		random_close_or_extended_family_member = {
			limit = {
				is_landed = yes
				NOT = {
					this = scope:crown
				}
			}
			save_scope_as = ruling_relative
		}
	}
	else_if = {
		limit = {
			dynasty ?= {
				any_dynasty_member = {
					is_landed = yes
				}
			}
		}
		dynasty ?= {
			random_dynasty_member = {
				limit = {
					is_landed = yes
				}
				save_scope_as = ruling_relative
			}
		}
	}
	scope:fleeing_winged_knight = {
		duel = {
			skill = prowess
			value = high_skill_rating

			50 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = 3.5
					min = -49
				}
				random_list = {
					70 = {
						trigger_event = agot_winged_knight.2004
					}
					25 = {
						trigger = { agot_valid_potential_nw_member = yes }
						agot_send_to_nightswatch_effect = { ACTOR = scope:fleeing_winged_knight NIGHTSWATCH_CANDIDATE = scope:fleeing_winged_knight }
						title:k_the_wall.holder = { add_courtier = scope:fleeing_winged_knight }
					}
					5 = {
						scope:ruling_relative = { trigger_event = agot_winged_knight.2003 }
					}
				}
			}
			50 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = -3.5
					min = -49
				}
				death = {
					death_reason = death_kg_fleeing
					killer = scope:crown
				}
			}
		}
	}
}

agot_winged_knight_destroy_artifacts = {
	if = {
		limit = {
			any_character_artifact = {
				has_variable = kingsguard_armor
			}
		}
		random_character_artifact = {
			limit = { has_variable = kingsguard_armor }
			destroy_artifact = this
		}
	}
}

agot_on_fired_from_winged_knight = {
	if = {
		limit = { is_alive = yes }
		fired_from_council_position_effect = yes
		remove_character_flag = blocked_from_leaving
		#remove_trait = winged_knight
	}
}

		}

agot_on_start_winged_knight_bodyguard = {
	scope:councillor_liege = {
		trigger_event = agot_winged_knight.2001
	}
}

agot_start_winged_knight_bodyguarding = {
	$TARGET$ = {
		set_relation_bodyguard = scope:councillor
	}
	scope:councillor = {
		set_variable = {
			name = agot_target_court
			value = scope:target_character
		}
		agot_start_councillor_travel_effect = yes
	}
}

agot_on_monthly_winged_knight_bodyguard = {
	scope:councillor = {
		# Add agot_target_court variable and winged_knight_bodyguard modifier to target if not present
		# This is mainly for savegame compatibility, could be removed at some point
		if = {
			limit = {
				NOT = { has_variable = agot_target_court }
			}
			set_variable = {
				name = agot_target_court
				value = scope:target_character
			}
		}
		every_relation = {
			type = bodyguard_target
			if = {
				limit = {
					NOT = { has_character_modifier = winged_knight_bodyguard }
				}
				add_character_modifier = winged_knight_bodyguard
			}
		}

		if = {
			limit = { is_available = yes }
			agot_start_councillor_travel_effect = yes
		}
	}
}

agot_on_cancel_winged_knight_bodyguard = {
	scope:councillor = {
		every_relation = {
			type = bodyguard_target
			remove_relation_bodyguard = scope:councillor
		}
		remove_variable = agot_target_court
		agot_cancel_councillor_travel_effect = yes
	}
}

			limit = {

					has_council_position = green_knight_lord_commander
					has_character_flag = lord_commander
				}
			}
			save_scope_as = kg_is_lord_commander
		}
	}

	scope:removed_kg = {
		if = {
			limit = {
				NOT = { exists = scope:dead_green_knight }
			}

			remove_trait = green_knight
			add_character_modifier = agot_former_green_knight
			agot_green_knight_destroy_artifacts = yes
			add_character_flag = can_fire_green_knight
			title:e_the_reach.holder ?= {
				if = {
					limit = {
						prev = { is_councillor_of = title:e_the_reach.holder }
					}
					fire_councillor = prev
				}
			}
		}
	}


	title:e_the_reach = {
		if = {
			limit = {
				exists = var:green_knight_lord_commander
				var:green_knight_lord_commander = scope:removed_kg
			}
			remove_variable = green_knight_lord_commander
		}
		else_if = {
			limit = {
				exists = var:green_knight_1
				var:green_knight_1 = scope:removed_kg
			}
			remove_variable = green_knight_1
		}
		else_if = {
			limit = {
				exists = var:green_knight_2
				var:green_knight_2 = scope:removed_kg
			}
			remove_variable = green_knight_2
		}
		else_if = {
			limit = {
				exists = var:green_knight_3
				var:green_knight_3 = scope:removed_kg
			}
			remove_variable = green_knight_3
		}
		else_if = {
			limit = {
				exists = var:green_knight_4
				var:green_knight_4 = scope:removed_kg
			}
			remove_variable = green_knight_4
		}
		else_if = {
			limit = {
				exists = var:green_knight_5

			remove_variable = green_knight_5
		}

				exists = var:green_knight_6

			}
			remove_variable = green_knight_6
		}
	}

	title:e_the_reach.holder ?= {
		if = {

				exists = scope:kg_is_lord_commander

					NOT = { exists = scope:dead_green_knight }

				limit = {
					any_councillor = {
						is_green_knight_trigger = yes
						has_trait = green_knight
						NOT = { this = scope:kg_is_lord_commander }

agot_remove_sun_knight_effect = {
	$KINGSGUARD$ = { save_scope_as = removed_kg }

	scope:removed_kg = {
		if = {
			limit = {
				OR = {
					has_council_position = sun_knight_lord_commander
					has_character_flag = lord_commander
				}
			}
			save_scope_as = kg_is_lord_commander
		}
	}

	scope:removed_kg = {
		if = {
			limit = {
				NOT = { exists = scope:dead_sun_knight }
			}

			remove_trait = sun_knight
			add_character_modifier = agot_former_sun_knight
			agot_sun_knight_destroy_artifacts = yes
			add_character_flag = can_fire_sun_knight
			title:e_dorne.holder ?= {
				if = {
					limit = {
						prev = { is_councillor_of = title:e_dorne.holder }
					}
					fire_councillor = prev
				}
			}
		}
	}


	title:e_dorne = {
		if = {
			limit = {
				exists = var:sun_knight_lord_commander
				var:sun_knight_lord_commander = scope:removed_kg
			}
			remove_variable = sun_knight_lord_commander
		}
		else_if = {
			limit = {
				exists = var:sun_knight_1
				var:sun_knight_1 = scope:removed_kg
			}
			remove_variable = sun_knight_1
		}
		else_if = {
			limit = {
				exists = var:sun_knight_2
				var:sun_knight_2 = scope:removed_kg
			}
			remove_variable = sun_knight_2
		}
		else_if = {
			limit = {
				exists = var:sun_knight_3
				var:sun_knight_3 = scope:removed_kg
			}
			remove_variable = sun_knight_3
		}
		else_if = {
			limit = {
				exists = var:sun_knight_4
				var:sun_knight_4 = scope:removed_kg
			}
			remove_variable = sun_knight_4
		}
		else_if = {
			limit = {
				exists = var:sun_knight_5
				var:sun_knight_5 = scope:removed_kg
			}
			remove_variable = sun_knight_5
		}
		else_if = {
			limit = {
				exists = var:sun_knight_6
				var:sun_knight_6 = scope:removed_kg
			}
			remove_variable = sun_knight_6
		}
	}

	title:e_dorne.holder ?= {
		if = {
			limit = {
				exists = scope:kg_is_lord_commander
			}
			if = {
				limit = {
					NOT = { exists = scope:dead_sun_knight }
				}
				destroy_title = title:d_sun_knight
			}
			if = {
				limit = {
					any_councillor = {
						is_sun_knight_trigger = yes
						has_trait = sun_knight
						NOT = { this = scope:kg_is_lord_commander }
					}
				}
				if = {
					limit = {
						is_ai = yes
					}
					trigger_event = {
						id = agot_sun_knight.9001
						days = 3
					}
				}
				else = {
					trigger_event = {
						id = agot_sun_knight.1010
						delayed = yes
					}
				}
			}
		}

		if = {
			limit = {
				is_ai = yes
			}
			trigger_event = {
				id = agot_sun_knight.9002
				days = 4
			}
		}
		else_if = {
			limit = {
				is_ai = no
				NOT = {

	scope:councillor = {
		# Add agot_target_court variable and sun_knight_bodyguard modifier to target if not present
		# This is mainly for savegame compatibility, could be removed at some point
		if = {
			limit = {
				NOT = { has_variable = agot_target_court }
			}
			set_variable = {
				name = agot_target_court
				value = scope:target_character
			}
		}
		every_relation = {
			type = bodyguard_target
			if = {
				limit = {
					NOT = { has_character_modifier = sun_knight_bodyguard }
				}
				add_character_modifier = sun_knight_bodyguard
			}
		}

		if = {
			limit = { is_available = yes }
			agot_start_councillor_travel_effect = yes
		}
	}
}

	}

agot_remove_storm_knight_effect = {
	$KINGSGUARD$ = { save_scope_as = removed_kg }

	scope:removed_kg = {
		if = {
			limit = {
				OR = {
					has_council_position = storm_knight_lord_commander
					has_character_flag = lord_commander
				}
			}
			save_scope_as = kg_is_lord_commander
		}
	}

	scope:removed_kg = {
		if = {
			limit = {
				NOT = { exists = scope:dead_storm_knight }
			}

			remove_trait = storm_knight
			add_character_modifier = agot_former_storm_knight
			agot_storm_knight_destroy_artifacts = yes
			add_character_flag = can_fire_storm_knight
			title:e_the_stormlands.holder ?= {
				if = {
					limit = {
						prev = { is_councillor_of = title:e_the_stormlands.holder }
					}
					fire_councillor = prev
				}
			}
		}
	}


	title:e_the_stormlands = {
		if = {
			limit = {
				exists = var:storm_knight_lord_commander
				var:storm_knight_lord_commander = scope:removed_kg
			}
			remove_variable = storm_knight_lord_commander
		}
		else_if = {
			limit = {
				exists = var:storm_knight_1
				var:storm_knight_1 = scope:removed_kg
			}
			remove_variable = storm_knight_1
		}
		else_if = {
			limit = {
				exists = var:storm_knight_2
				var:storm_knight_2 = scope:removed_kg
			}
			remove_variable = storm_knight_2
		}
		else_if = {
			limit = {
				exists = var:storm_knight_3
				var:storm_knight_3 = scope:removed_kg
			}
			remove_variable = storm_knight_3
		}
		else_if = {
			limit = {
				exists = var:storm_knight_4
				var:storm_knight_4 = scope:removed_kg
			}
			remove_variable = storm_knight_4
		}
		else_if = {
			limit = {
				exists = var:storm_knight_5
				var:storm_knight_5 = scope:removed_kg
			}
			remove_variable = storm_knight_5
		}
		else_if = {
			limit = {
				exists = var:storm_knight_6
				var:storm_knight_6 = scope:removed_kg
			}
			remove_variable = storm_knight_6
		}
	}

	title:e_the_stormlands.holder ?= {
		if = {
			limit = {
				exists = scope:kg_is_lord_commander
			}
			if = {
				limit = {
					NOT = { exists = scope:dead_storm_knight }
				}
				destroy_title = title:d_storm_knight
			}
			if = {
				limit = {
					any_councillor = {
						is_storm_knight_trigger = yes
						has_trait = storm_knight
						NOT = { this = scope:kg_is_lord_commander }
					}
				}
				if = {
					limit = {
						is_ai = yes
					}

						id = agot_storm_knight.9001
						days = 3
					}

			limit = {

			}

				id = agot_storm_knight.9002
				days = 4
			}

			limit = {
				NOT = { has_variable = agot_target_court }
			}
			set_variable = {
				name = agot_target_court
				value = scope:target_character
			}
		}
		every_relation = {
			type = bodyguard_target
			if = {
				limit = {
					NOT = { has_character_modifier = storm_knight_bodyguard }
				}
				add_character_modifier = storm_knight_bodyguard
			}
		}

		}
	}

		# This is mainly for savegame compatibility, could be removed at some point
		if = {
			limit = {
				NOT = { has_variable = agot_target_court }
			}
			set_variable = {
				name = agot_target_court
				value = scope:target_character
			}
		}
		every_relation = {
			type = bodyguard_target
			if = {
				limit = {
					NOT = { has_character_modifier = saltguard_bodyguard }
				}
				add_character_modifier = saltguard_bodyguard
			}
		}

		if = {
			limit = { is_available = yes }
			agot_start_councillor_travel_effect = yes
		}
	}

			type = agot_winter_wolf_lord_commander_memory
			participants = {
				king = $KING$
			}
		}
		add_character_flag = lord_commander
	}

	$KING$ = {
		primary_title = {
			if = {
				limit = {
					exists = var:winter_wolf_1
					var:winter_wolf_1 = $COMMANDER$
				}
				remove_variable = winter_wolf_1
			}
			else_if = {
				limit = {
					exists = var:winter_wolf_2
					var:winter_wolf_2 = $COMMANDER$
				}
				remove_variable = winter_wolf_2
			}
			else_if = {
				limit = {
					exists = var:winter_wolf_3
					var:winter_wolf_3 = $COMMANDER$
				}
				remove_variable = winter_wolf_3
			}
			else_if = {
				limit = {
					exists = var:winter_wolf_4
					var:winter_wolf_4 = $COMMANDER$
				}
				remove_variable = winter_wolf_4
			}
			else_if = {
				limit = {
					exists = var:winter_wolf_5
					var:winter_wolf_5 = $COMMANDER$
				}
				remove_variable = winter_wolf_5
			}
			else_if = {
				limit = {
					exists = var:winter_wolf_6
					var:winter_wolf_6 = $COMMANDER$
				}
				remove_variable = winter_wolf_6
			}

			set_variable = {
				name = winter_wolf_lord_commander
				value = $COMMANDER$
			}
		}


		assign_councillor_type = {
			type = winter_wolf_lord_commander
			target = $COMMANDER$
		}
	}
	if = {
		limit = {
			$COMMANDER$ = { is_lowborn = yes }
		}
		save_temporary_scope_as = lc_is_lowborn
	}

	hidden_effect = {
		title:d_winter_wolf = {
			copy_title_history = title:d_dummy_winter_wolf
		}
	}

	create_title_and_vassal_change = {
		type = granted
		save_scope_as = change
	}
	title:d_winter_wolf = {
		change_title_holder = {
			holder = $COMMANDER$
			change = scope:change
		}
	}

	resolve_title_and_vassal_change = scope:change

	if = {
		limit = {
			NOT = {
				$COMMANDER$.liege = $KING$
			}
		}
		create_title_and_vassal_change = {
			type = swear_fealty
			save_scope_as = fealty_change
			add_claim_on_loss = no
		}
		title:d_winter_wolf.holder = {
			change_liege = {
				liege = $KING$
				change = scope:fealty_change
			}
		}
		resolve_title_and_vassal_change = scope:fealty_change
	}

	if = {
		limit = { exists = scope:lc_is_lowborn }
		$COMMANDER$ = { set_to_lowborn = yes }
	}

	title:d_winter_wolf = {
		set_coa = d_winter_wolf
	}

	hidden_effect = {
		title:d_dummy_winter_wolf = {
			copy_title_history = title:d_winter_wolf
		}
	}

	$KING$ = { trigger_event = agot_winter_wolf.9007 } #grab back the lord commander you yeeted
}

agot_remove_winter_wolf_effect = {
	$KINGSGUARD$ = { save_scope_as = removed_kg }

	scope:removed_kg = {
		if = {
			limit = {
				OR = {
					has_council_position = winter_wolf_lord_commander
					has_character_flag = lord_commander
				}
			}
			save_scope_as = kg_is_lord_commander
		}
	}

	scope:removed_kg = {
		if = {
			limit = {
				NOT = { exists = scope:dead_winter_wolf }
			}

			remove_trait = winter_wolf
			add_character_modifier = agot_former_winter_wolf
			agot_winter_wolf_destroy_artifacts = yes
			add_character_flag = can_fire_winter_wolf
			title:e_the_north.holder ?= {
				if = {
					limit = {
						prev = { is_councillor_of = title:e_the_north.holder }
					}
					fire_councillor = prev
				}
			}
		}
	}


	title:e_the_north = {
		if = {
			limit = {
				exists = var:winter_wolf_lord_commander
				var:winter_wolf_lord_commander = scope:removed_kg
			}
			remove_variable = winter_wolf_lord_commander
		}
		else_if = {
			limit = {
				exists = var:winter_wolf_1
				var:winter_wolf_1 = scope:removed_kg
			}
			remove_variable = winter_wolf_1
		}
		else_if = {
			limit = {
				exists = var:winter_wolf_2
				var:winter_wolf_2 = scope:removed_kg
			}
			remove_variable = winter_wolf_2
		}
		else_if = {
			limit = {
				exists = var:winter_wolf_3
				var:winter_wolf_3 = scope:removed_kg
			}
			remove_variable = winter_wolf_3
		}
		else_if = {
			limit = {
				exists = var:winter_wolf_4
				var:winter_wolf_4 = scope:removed_kg
			}
			remove_variable = winter_wolf_4
		}
		else_if = {
			limit = {
				exists = var:winter_wolf_5
				var:winter_wolf_5 = scope:removed_kg
			}
			remove_variable = winter_wolf_5
		}
		else_if = {
			limit = {
				exists = var:winter_wolf_6
				var:winter_wolf_6 = scope:removed_kg
			}
			remove_variable = winter_wolf_6
		}
	}

	title:e_the_north.holder ?= {
		if = {
			limit = {
				exists = scope:kg_is_lord_commander
			}
			if = {
				limit = {
					NOT = { exists = scope:dead_winter_wolf }
				}
				destroy_title = title:d_winter_wolf
			}
			if = {
				limit = {
					any_councillor = {
						is_winter_wolf_trigger = yes
						has_trait = winter_wolf
						NOT = { this = scope:kg_is_lord_commander }
					}
				}
				if = {
					limit = {
						is_ai = yes
					}
					trigger_event = {
						id = agot_winter_wolf.9001
						days = 3
					}
				}
				else = {
					trigger_event = {
						id = agot_winter_wolf.1010
						delayed = yes
					}
				}
			}
		}

		if = {
			limit = {
				is_ai = yes
			}
			trigger_event = {
				id = agot_winter_wolf.9002
				days = 4
			}
		}
		else_if = {
			limit = {
				is_ai = no
				NOT = {
					has_character_flag = skip_kg_replacement_event_flag
				}
			}
			trigger_event = agot_winter_wolf.1001
		}
	}
}

agot_winter_wolf_fled_effect = {
	agot_remove_winter_wolf_effect = { KINGSGUARD = scope:recipient }
	add_character_modifier = winter_wolf_oathbreaker

	scope:recipient = {
		save_scope_as = fleeing_winter_wolf
	}

	scope:actor = {
		save_scope_as = crown
	}

	scope:crown = {
		add_opinion = {
			target = scope:fleeing_winter_wolf
			modifier = broke_vows_opinion
		}
	}

	if = {
		limit = {
			any_parent = {
				is_landed = yes
				NOT = {
					this = scope:crown
				}
			}
		}
		random_parent = {
			limit = {
				is_landed = yes
				NOT = {
					this = scope:crown
				}
			}
			save_scope_as = ruling_relative
		}
	}
	else_if = {
		limit = {
			any_close_family_member = {
				is_landed = yes
				NOT = {
					this = scope:crown
				}
			}
		}
		random_close_family_member = {
			limit = {
				is_landed = yes
				NOT = {
					this = scope:crown
				}
			}
			save_scope_as = ruling_relative
		}
	}
	else_if = {
		limit = {
			any_close_or_extended_family_member = {
				is_landed = yes
				NOT = {
					this = scope:crown
				}
			}
		}
		random_close_or_extended_family_member = {
			limit = {
				is_landed = yes
				NOT = {
					this = scope:crown
				}
			}
			save_scope_as = ruling_relative
		}
	}
	else_if = {
		limit = {
			dynasty ?= {
				any_dynasty_member = {
					is_landed = yes
				}
			}
		}
		dynasty ?= {
			random_dynasty_member = {
				limit = {
					is_landed = yes
				}
				save_scope_as = ruling_relative
			}
		}
	}
	scope:fleeing_winter_wolf = {
		duel = {
			skill = prowess
			value = high_skill_rating

			50 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = 3.5
					min = -49
				}
				random_list = {
					70 = {
						trigger_event = agot_winter_wolf.2004
					}
					25 = {
						trigger = { agot_valid_potential_nw_member = yes }
						agot_send_to_nightswatch_effect = { ACTOR = scope:fleeing_winter_wolf NIGHTSWATCH_CANDIDATE = scope:fleeing_winter_wolf }
						title:k_the_wall.holder = { add_courtier = scope:fleeing_winter_wolf }
					}
					5 = {
						scope:ruling_relative = { trigger_event = agot_winter_wolf.2003 }
					}
				}
			}
			50 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = -3.5
					min = -49
				}
				death = {
					death_reason = death_kg_fleeing
					killer = scope:crown
				}
			}
		}
	}
}

	if = {
		limit = {

			}
		}

			destroy_artifact = this
		}
	}
}

agot_on_start_winter_wolf_bodyguard = {
	scope:councillor_liege = {
		trigger_event = agot_winter_wolf.2001
	}
}

agot_start_winter_wolf_bodyguarding = {
	$TARGET$ = {
		set_relation_bodyguard = scope:councillor
	}
	scope:councillor = {
		set_variable = {
			name = agot_target_court
			value = scope:target_character
		}
		agot_start_councillor_travel_effect = yes
	}
}

agot_on_monthly_winter_wolf_bodyguard = {
	scope:councillor = {
		# Add agot_target_court variable and winter_wolf_bodyguard modifier to target if not present
		# This is mainly for savegame compatibility, could be removed at some point
		if = {
			limit = {
				NOT = { has_variable = agot_target_court }
			}
			set_variable = {
				name = agot_target_court
				value = scope:target_character
			}
		}
		every_relation = {
			type = bodyguard_target
			if = {
				limit = {
					NOT = { has_character_modifier = winter_wolf_bodyguard }
				}
				add_character_modifier = winter_wolf_bodyguard
			}
		}

		if = {
			limit = { is_available = yes }
			agot_start_councillor_travel_effect = yes
		}
	}
}

agot_on_cancel_winter_wolf_bodyguard = {
	scope:councillor = {
		every_relation = {
			type = bodyguard_target
			remove_relation_bodyguard = scope:councillor
		}
		remove_variable = agot_target_court
		agot_cancel_councillor_travel_effect = yes
	}
}
