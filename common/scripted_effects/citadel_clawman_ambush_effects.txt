citadel_create_clawman_ambush = {
	create_character = {
		name = clawman_leader_name
		gender = male
		template = clawman_ambusher_template
		save_scope_as = clawman
		location = scope:target_location
		culture = culture:clawman
		faith = faith:old_gods_wnw
	}
	create_dynamic_title = {
		tier = duchy
		name = clawman_duchy_title
	}
	create_title_and_vassal_change = {
		type = created
		save_scope_as = change
		add_claim_on_loss = no
	}
	scope:new_title = {
		set_capital_county = $TARGET$.capital_county
		set_landless_title = yes
		set_destroy_on_succession = yes
		set_delete_on_destroy = yes
		set_no_automatic_claims = yes
		set_definitive_form = yes
		set_can_be_named_after_dynasty = no
		change_title_holder = {
			holder = scope:clawman
			change = scope:change
		}
	}
	resolve_title_and_vassal_change = scope:change
}

citadel_spawn_clawman_army = {
	scope:clawman = {
		hidden_effect = { add_gold = 100 }
		start_war = {
			cb = clawman_ambush_cb
			target = $TARGET$
			target_title = $TARGET$.primary_title
		}
		if = {
			limit = { exists = scope:target_location }
			spawn_army = {
				name = clawman_army_name
				levies = 0
				men_at_arms = {
					type = marsh_walkers
					stacks = 16
				}
				location = scope:target_location
			}
		}
		else = {
			spawn_army = {
				name = clawman_army_name
				levies = 0
				men_at_arms = {
					type = marsh_walkers
					stacks = 16
				}
				location = $TARGET$.location
			}
		}
	}
}
