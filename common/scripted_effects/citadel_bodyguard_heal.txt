### Citadel Bodyguard orders - self-heal stuck order slots. - Kizan
# If a member died BEFORE the on_death maintenance fix existed, their slot variable on the empire title was never
# cleared, so the order looked "full" and replacements were blocked. This clears any slot variable whose target is
# dead, purged, or no longer a member. Called from on_death, so existing saves heal within days of play (no console).

citadel_clear_dead_slot_effect = {
	if = {
		limit = {
			$TITLE$ = { exists = var:$SLOT$ }
			OR = {
				NOT = { exists = $TITLE$.var:$SLOT$ }
				$TITLE$.var:$SLOT$ = { is_alive = no }
				$TITLE$.var:$SLOT$ = { NOT = { has_trait = $TRAIT$ } }
			}
		}
		$TITLE$ = { remove_variable = $SLOT$ }
	}
}
