asoiaf_north_contract_effect = {
	vassal_contract_set_obligation_level = {
		type = feudal_government_taxes
		level = 1
	}
	vassal_contract_set_obligation_level = {
		type = feudal_government_levies
		level = 0
	}
	vassal_contract_set_obligation_level = {
		type = title_revocation_rights
		level = 1
	}
	if = { #ONLY set to Marcher if the special contract was reset to NORMAL (if the player gave made it something else, it should stay that way)
		limit = {
			NOR = {
				vassal_contract_has_flag = has_march_contract
				vassal_contract_has_flag = has_palatinate_contract
				vassal_contract_has_flag = has_scutage_contract
			}
		}
		vassal_contract_set_obligation_level = {
			type = special_contract
			level = 2 #Marcher
		}
	}
	if = {
		limit = {
			NOT = { root.faith = root.liege.faith }
		}
		vassal_contract_set_obligation_level = {
			type = religious_rights
			level = 1
		}
	}
}

			}
		}
		vassal_contract_set_obligation_level = {
			type = special_contract
