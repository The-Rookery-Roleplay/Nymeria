		scope:recipient = {

				AND = {
					is_child_of = scope:actor
					is_landed = no
					is_courtier_of = scope:actor
				}
			}
			is_male = yes
			age >= 16
		}
		scope:actor = {
			NOR = {
				has_trait = just
				has_trait = compassionate
				custom_tooltip = {
					text = amsb_WE_ARE_FRIENDS
					has_relation_friend = scope:recipient
					has_relation_best_friend = scope:recipient
				}
			}
			OR = {
				has_trait = callous
				has_trait = sadistic
				has_trait = arbitrary
				has_relation_rival = scope:recipient
				has_relation_nemesis = scope:recipient
			}
		}
	}
	is_valid_showing_failures_only = {
		scope:recipient = {
			NOR = {
				is_allied_to = scope:actor
				has_strong_hook = scope:actor
			}
		}
	}

	can_send = {
		scope:recipient = {
			NOT = { this = scope:actor }
			OR = {
				# is_imprisoned_by = scope:actor
				AND = {
					is_child_of = scope:actor
					is_landed = no
					is_courtier_of = scope:actor
					scope:actor = {
						OR = {
							has_trait = arrogant
							has_trait = callous
							has_trait = sadistic
							has_trait = arbitrary
							has_relation_rival = scope:recipient
							has_relation_nemesis = scope:recipient
						}
					}
				}
			}
			is_human = yes
			is_male = yes
			age >= 16
		}
		scope:actor = {
			is_ai = no
			is_imprisoned = no
			is_human = yes
		}
	}

	on_auto_accept = { }

			multiplier = 0.5

	on_accept = {
		scope:recipient = {
			agot_add_to_nightswatch_effect = yes
			title:k_the_wall.holder = {
				trigger_event = agot_nights_watch.0100
				add_courtier = scope:recipient
			}
			if = {
				limit = {
					religion = religion:the_pact_religion
				}
				add_prestige = 250
				add_stress = medium_stress_impact_loss
			}
			else = {
				add_prestige = -250
				add_stress = major_stress_impact_gain
			}
		}
		scope:actor = {
			add_prestige = -500
			stress_impact = {
				callous = minor_stress_impact_loss
				arbitrary = medium_stress_impact_loss
				sadistic = major_stress_impact_loss
			}
		}
	}

	on_decline = {

				SC_INITIATOR = scope:actor
				SC_ATTACKER = scope:actor
				SC_DEFENDER = scope:recipient

				FIXED = no
				LOCALE = terrain_scope
				OUTPUT_EVENT = single_combat.0001
				INVALIDATION_EVENT = single_combat.0001	#In a real event, this would obviously be a different even to the output.
			}
		}
		scope:actor = {
			add_prestige = -1000
			stress_impact = {
				callous = minor_stress_impact_loss
				arbitrary = medium_stress_impact_gain
				sadistic = major_stress_impact_gain
			}
		}
