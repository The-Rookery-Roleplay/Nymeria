# A petty Dornish duke proclaims himself king, creating a titular ("de facto")
# kingdom named after his current duchy. No de jure shifting happens, so the
# realm's de jure structure is untouched - he simply gains a crown so he can
# stand against the other kings of Dorne. Everyone in Dorne is notified.

nymeria_self_proclaim_kingdom_decision = {
	picture = {
		reference = "gfx/interface/illustrations/decisions/decision_found_kingdom.dds"
	}
	decision_group_type = major

	ai_check_interval = 60

	desc = nymeria_self_proclaim_kingdom_decision_desc
	selection_tooltip = nymeria_self_proclaim_kingdom_decision_tooltip
	confirm_text = nymeria_self_proclaim_kingdom_decision_confirm

	cost = {
		gold = 100
		prestige = 150
	}

	is_shown = {
		highest_held_title_tier = tier_duchy
		is_independent_ruler = yes
		# Must actually rule land inside Dorne. Stops a wandering landless adventurer who
		# holds only a titular duchy, or some lord from beyond the red mountains, from
		# crowning himself a king of Dorne. - Kizan
		is_landed = yes
		capital_county = { target_is_de_jure_liege_or_above = title:e_dorne }
		# Santagar is flagged at game start (story on) so he stays a duke for the
		# Broken Arm war instead of crowning himself. - Kizan
		NOT = { has_character_flag = nymeria_cannot_proclaim_kingdom }
	}

	is_valid = {
		highest_held_title_tier = tier_duchy
		is_independent_ruler = yes
		is_landed = yes
		capital_county = { target_is_de_jure_liege_or_above = title:e_dorne }
		is_imprisoned = no
	}

	ai_potential = {
		always = yes
	}

	ai_will_do = {
		base = 0
		modifier = {
			add = 100
			OR = {
				has_trait = ambitious
				has_trait = stubborn
				has_trait = brave
			}
			NOT = { dynasty = dynasty:dynn_Martell }
		}
	}

	effect = {
		custom_tooltip = nymeria_self_proclaim_kingdom_decision_effect_tt
		custom_tooltip = nymeria_self_proclaim_kingdom_decision_cb_tt

		save_scope_as = new_king
		primary_title = {
			save_scope_as = old_title
		}

		# Title creation is hidden so the effect tooltip doesn't show a
		# half-resolved "None of becomes your Primary Title" line - the
		# dynamic title doesn't exist until the decision actually runs.
		hidden_effect = {
			# Create the titular kingdom, named/coloured after the old duchy.
			# No set_de_jure_liege_title -> the realm's de jure map is untouched.
			create_dynamic_title = {
				tier = kingdom
				name = NEW_CREATED_TITLE_NAME
			}
			create_title_and_vassal_change = {
				type = created
				save_scope_as = change
				add_claim_on_loss = no
			}
			scope:new_title = {
				set_color_from_title = scope:old_title
				set_coa = scope:old_title
				change_title_holder = {
					holder = root
					change = scope:change
				}
			}
			resolve_title_and_vassal_change = scope:change
			set_primary_title_to = scope:new_title

			# Word spreads across Dorne that another petty lord has crowned himself -
			# the new king receives the proclamation too.
			every_ruler = {
				limit = {
					capital_county = { target_is_de_jure_liege_or_above = title:e_dorne }
				}
				trigger_event = nymeria_dorne_events.0008
			}
		}
	}
}
