### Citadel Bodyguard orders - guard governments for Winged Knights (Vale) + Green Knights (Reach). - Kizan
# Faithful copies of AGOT kingsguard_government, gated to each order's own title/trait so that
# change_government = winged_knight_government / green_knight_government (set in the join events) resolves.

winged_knight_government = {
	government_rules = {
		create_cadet_branches = no
		court_generate_spouses = no
		council = no
		religious = no
		rulers_should_have_dynasty = no
		dynasty_named_realms = no
		allow_accolades = no
	}

	court_generate_commanders = no

	primary_holding = castle_holding

	can_get_government = {
		is_landed = no
		OR = {
			has_title = title:d_winged_knight
			has_trait = winged_knight
		}
	}

	ai = {
		arrange_marriage = no
		use_goals = no
		use_decisions = no
		perform_religious_reformation = no
	}

	flags = {
		government_is_kg
		government_is_settled
		government_is_default
	}

	color = rgb{ 214 245 250 }
}
