### Citadel Bodyguard orders (titles) - duplicated from AGOT kingsguard for Vale (Winged Knights) + Reach (Green Knights). - Kizan

# script constants the title blocks reference (must be defined in-file)
@correct_culture_primary_score = 100
@better_than_the_alternatives_score = 50
@always_primary_score = 1000

##### WINGED KNIGHTS (Vale) #####
d_winged_knight = {
	color = { 255 249 198 }

	definite_form = yes
	ruler_uses_title_name = no
	landless = yes
	can_create = { always = no }
	destroy_on_succession = yes

	capital = c_kings_landing

	ai_primary_priority = {
		add = @always_primary_score
	}

	no_automatic_claims = yes
	de_jure_drift_disabled = yes
}
