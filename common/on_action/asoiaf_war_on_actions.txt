on_war_won_defender = {
	on_actions = {
		asoiaf_forge_krakenfall_on_action
	}
}

asoiaf_forge_krakenfall_on_action = {
	trigger = {
		scope:war.primary_attacker.dynasty = dynasty:dynn_Greyjoy
		scope:war.primary_attacker = { has_title = title:e_the_iron_islands }
		NOT = { scope:war.primary_defender.dynasty = dynasty:dynn_Targaryen } #King can be any dynasty except Targaryens: I don't want this event to be annoying spam for every person's playthrough as king
		scope:war.primary_defender = { has_title = title:h_the_iron_throne } #The recipient is King of the Seven Kingdoms
		NOT = { exists = global_var:asoiaf_krakenfall_is_forged }
		has_game_rule = asoiaf_new_clothes_stormlands_rule_on
	}
	effect = {
		scope:war.primary_defender = { trigger_event = { id = asoiaf_maintenance_events.0010 days = 30 } }
		set_global_variable = asoiaf_krakenfall_is_forged #so this can't trigger twice
	}
}
