on_building_completed = {
	on_actions = {
		harlaw_building_check_on_actions
	}
}

harlaw_building_check_on_actions = {
	effect = {
		#save the county holder as the character in case the construction wasn't started through GUI or by the AI 
		if = {
			limit = {
				NOT = { exists = scope:character }
			}
			root.county.holder ?= {
				save_scope_as = character
			}
		}
		if = {
			limit = {
				this = province:2819
				has_holding_type = castle_holding
				has_special_building_slot = no
				exists = scope:character.dynasty
				scope:character = {
					dynasty = dynasty:dynn_Harlaw
				}
			}
			add_special_building_slot = agot_ten_towers_01
		}
		if = {
			limit = {
				this = province:2819
				has_holding_type = castle_holding
				has_special_building_slot = no
				exists = scope:character.dynasty
				scope:character = {
					NOT = { dynasty = dynasty:dynn_Harlaw }
				}
			}
			add_special_building_slot = harlaw_mines_01
		}
	}
}
