nymeria_laamp_events.0002 = { # Settle Dorne?
    hidden = no
    type = character_event
    title = nymeria_laamp_events.0002.t
    desc = nymeria_laamp_events.0002.desc
    theme = realm

    trigger = {
        is_character = character:Nymeria_1
        has_game_rule = nymeria_dorne_story_on
    }

    left_portrait = {
        character = root
        animation = thinking
    }

    immediate = {
        set_character_flag = nymeria_story_002_fired
    }

    option = {
        name = nymeria_laamp_events.0002.a
        ai_chance = { base = 100 }
        trigger_event = {
            id = nymeria_laamp_events.0003
        }
        custom_tooltip = nymeria_laamp_events0002a_popup
    }

    option = {
        name = nymeria_laamp_events.0002.b
        ai_chance = { base = 0 }
        set_character_flag = nymeria_story_cycle_ended
        custom_tooltip = nymeria_laamp_events0002b_popup
    }
}
