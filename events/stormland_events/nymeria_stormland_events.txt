# Initial flavor event - fires on Durrandon_62
nymeria_stormlands.0001 = {
    type = character_event
    title = nymeria_stormlands.0001.t
    desc = nymeria_stormlands.0001.desc
    theme = war
    left_portrait = {
        character = scope:storm_king
        animation = rage
    }

    trigger = {
        has_game_rule = nymeria_stormlands_story_on
        this = character:Durrandon_62
        NOT = { has_character_flag = stormlands_war_pending }
    }

    immediate = {
        save_scope_as = storm_king
        remove_character_flag = stormlands_war_pending
        title:k_the_hook.holder = { trigger_event = nymeria_stormlands.0003 }
        title:k_the_blackwater.holder = { trigger_event = nymeria_stormlands.0003 }
        title:k_duskwashes.holder = { trigger_event = nymeria_stormlands.0003 }
    }

    option = {
        name = nymeria_stormlands.0001.a   # "Ours is the Fury"
        add_character_flag = stormlands_war_pending
        custom_tooltip = nymeria_stormlands0001a_popup
        ai_chance = { base = 100 }
    }
}

        ai_chance = { base = 0 }
