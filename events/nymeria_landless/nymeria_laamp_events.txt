nymeria_laamp_events.0007 = { # Martell weighs Toland's alliance against the Rhoynar
    type = character_event
    title = nymeria_laamp_events.0007.t
    desc = nymeria_laamp_events.0007.desc
    theme = realm

    left_portrait = {
        character = root
        animation = thinking
    }
    right_portrait = {
        character = character:Toland_200
        animation = personality_bold
    }

    trigger = {
        this = character:Martell_23
    }

    option = { # Accept - crush the Rhoynar together
        name = nymeria_laamp_events.0007.a
        ai_chance = { base = 95 }

        # Toland declares the war; Martell joins as co-attacker. - Kizan
        character:Toland_200 = {
            start_war = {
                cb = nymeria_crush_rhoynar
                target = character:Nymeros_1
                target_title = title:c_torchwood
            }
            every_character_war = {
                limit = { using_cb = nymeria_crush_rhoynar }
                add_attacker = character:Martell_23
            }
        }
    }

    option = { # Decline - let the petty lord drown alone
        name = nymeria_laamp_events.0007.b
        ai_chance = { base = 5 }

        # A few days on, Nymeria may take the fight to one of them. The flag marks
        # the no-alliance path so she can be re-offered the second target. - Kizan
        character:Nymeros_1 = {
            add_character_flag = nymeria_route_b_active
            trigger_event = {
                id = nymeria_laamp_events.0008
                days = 5
            }
        }
    }
}

nymeria_laamp_events.0009 = { # Nymeria sends letters of dominance up the Greenblood
    type = character_event
    title = nymeria_laamp_events.0009.t
    desc = nymeria_laamp_events.0009.desc
    theme = realm

    left_portrait = {
        character = root
        animation = personality_bold
    }

    trigger = {
        this = character:Nymeros_1
    }

    immediate = {
        save_scope_as = rhoynar_queen
    }

    option = { # Let the spears speak for the ink
        name = nymeria_laamp_events.0009.a

        # Gather the lord of each Greenblood duchy by TITLE (not a hardcoded ID).
        # An empty duchy is skipped; the list is a set, so a lord who holds two
        # of the three duchies is only summoned once. - Kizan
        if = {
            limit = {
                exists = title:d_godsgrace.holder
                title:d_godsgrace.holder = { is_independent_ruler = yes }
            }
            title:d_godsgrace.holder = { add_to_temporary_list = greenblood_lords }
        }
        if = {
            limit = {
                exists = title:d_salt_shore.holder
                title:d_salt_shore.holder = { is_independent_ruler = yes }
            }
            title:d_salt_shore.holder = { add_to_temporary_list = greenblood_lords }
        }
        if = {
            limit = {
                exists = title:d_lemonwood.holder
                title:d_lemonwood.holder = { is_independent_ruler = yes }
            }
            title:d_lemonwood.holder = { add_to_temporary_list = greenblood_lords }
        }
        every_in_list = {
            list = greenblood_lords
            trigger_event = { id = nymeria_laamp_events.0010 days = 2 }
        }
    }
}

nymeria_laamp_events.0010 = { # A Greenblood lord answers the Rhoynar's letter
    type = character_event
    title = nymeria_laamp_events.0010.t
    desc = nymeria_laamp_events.0010.desc
    theme = realm

    left_portrait = {
        character = root
        animation = anger
    }
    right_portrait = {
        character = scope:rhoynar_queen
        animation = personality_bold
    }

    option = { # Bend the knee to the Rhoynar
        name = nymeria_laamp_events.0010.a
        # A crowned lord will never bow - this option is barred to kings. - Kizan
        trigger = {
            NOT = { any_held_title = { tier = tier_kingdom } }
        }
        ai_chance = { base = 50 }

        create_title_and_vassal_change = {
            type = swear_fealty
            save_scope_as = greenblood_fealty
        }
        root = {
            change_liege = {
                liege = scope:rhoynar_queen
                change = scope:greenblood_fealty
            }
        }
        resolve_title_and_vassal_change = scope:greenblood_fealty
    }

    option = { # Defy the spears
        name = nymeria_laamp_events.0010.b
        ai_chance = { base = 50 }

        # Nymeria invades - the CB breaks the lord's crown first if he wears one. - Kizan
        scope:rhoynar_queen = {
            start_war = {
                cb = nymeria_rhoynish_invasion
                target = root
                target_title = root.primary_title
            }
        }
    }
}
