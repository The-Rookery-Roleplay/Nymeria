scripted_trigger agot_eligible_knight_to_champion_trial_by_combat_trigger = {
	is_ruler = no
	can_be_combatant_based_on_gender_trigger = { ARMY_OWNER = scope:needs_champion }
	NOT = { has_trait = maester }
	NOT = { has_trait = septon }
	eligible_knight_to_champion_trial_by_combat_trigger = yes

	trigger_if = {
		limit = { title:h_the_iron_throne.holder ?= { is_ai = no } }
		NOT = { has_trait = kingsguard }
	}

	trigger_if = {
		limit = {
			NOT = { has_trait = nightswatch }
		}
		NOT = { has_trait = nightswatch }
	}

	#Would agree to it
	NOT = { has_trait = craven }
	NOT = { has_relation_rival = scope:needs_champion }
	trigger_if = {
		limit = { any_heir_title = {} }
		OR = {
			has_relation_friend = scope:needs_champion
			has_relation_lover = scope:needs_champion
		}
	}
	trigger_if = {
		limit = {
			NOT = { liege_or_court_owner ?= scope:needs_champion }
		}
		save_temporary_opinion_value_as = { name = target_opinion target = scope:needs_champion }
		scope:target_opinion >= medium_positive_opinion
	}
	trigger_else = {
		save_temporary_opinion_value_as = { name = target_opinion target = scope:needs_champion }
		scope:target_opinion >= 0
	}
}

	option = {

		ai_chance = {
			base = 0

			#Prowess-Rationality-Boldness Calculation.
			modifier = {
				add = {
					value = 0
					add = {
						value = prowess
						divide = 2
						min = 1
					}
					multiply = {
						value = ai_rationality
						#no 0s.
						if = {
							limit = { ai_rationality = 0 }
							add = -1
						}
					}
					divide = {
						value = 20
						#If ai_rationality is around 0, not dividing at all is fine, so it won't invalidate ai_boldness calculations.
						if = {
							limit = {
								AND = {
									ai_rationality >= -10
									ai_rationality <= 10
								}
							}
							divide = 20
						}
					}
					#Powerful and Irrational characters will decide based on their Boldness (in negative to balance out their negative ai_rationality).
					if = {
						limit = {
							AND = {
								prowess >= 25
								ai_rationality <= 0
							}
						}
						multiply = {
							value = ai_boldness
							multiply = -0.01
						}
					}
					#Powerless and Irrational characters will decide based on half of their Boldness (in negative to balance out their negative ai_rationality).
					else_if = {
						limit = {
							AND = {
								prowess < 25
								ai_rationality <= 0
							}
						}
						multiply = {
							value = ai_boldness
							multiply = -0.02
						}
					}
					#Rational characters will decide based on their Boldness (eg. if they are Powerless they can still go for it if they are brave enough).
					else = {
						multiply = {
							value = ai_boldness
							multiply = 0.01
						}
					}
				}
			}

			#Relation to Prowess.
			modifier = {
				factor = {
					value = 1
					subtract = {
						value = prowess
						subtract = 25
						multiply = 0.2
					}
				}
			}

			#King has stronger knights.
			modifier = {
				add = 100

				highest_held_title_tier >= tier_empire
				any_knight = {
					age > 15
					can_be_combatant_based_on_gender_trigger = { ARMY_OWNER = root }
					is_physically_able_ai = yes
					prowess >= scope:recipient.prowess
				}
			}
		}
	}
